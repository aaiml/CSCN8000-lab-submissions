# finding wcss value for different number of clusters

from sklearn.cluster import KMeans

wcss = []
for i in range(1, 11):
    kmeans = KMeans(n_clusters=i, init='k-means++', max_iter=300, n_init=25, random_state=16)
    kmeans.fit(data)
    wcss.append(kmeans.inertia_)
