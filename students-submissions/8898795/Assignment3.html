import numpy as np
import pandas as pd
from sklearn import datasets
from sklearn.cluster import KMeans
import matplotlib.pyplot as plt

# Load the Iris dataset
iris = datasets.load_iris()
data = iris.data  # The feature matrix
target = iris.target  # The target labels

# Create a DataFrame with petal width and length features
petal_data = pd.DataFrame(data[:, 2:4], columns=['Petal Length', 'Petal Width'])

# Perform KMeans clustering
num_clusters = 3  # Number of clusters to form
kmeans = KMeans(n_clusters=num_clusters)
kmeans.fit(petal_data)

# Add cluster labels to the DataFrame
petal_data['Cluster'] = kmeans.labels_

# Visualize the clusters
plt.figure(figsize=(8, 6))
plt.scatter(petal_data['Petal Length'], petal_data['Petal Width'], c=petal_data['Cluster'], cmap='viridis')
plt.scatter(kmeans.cluster_centers_[:, 0], kmeans.cluster_centers_[:, 1], s=300, c='red', marker='X', label='Cluster Centers')
plt.xlabel('Petal Length')
plt.ylabel('Petal Width')
plt.title('Clustering of Iris Flowers based on Petal Features')
plt.legend()
plt.show()

# Calculate inertia for different values of k
inertia = []
for k in range(1, 11):
    kmeans = KMeans(n_clusters=k)
    kmeans.fit(scaled_petal_data)
    inertia.append(kmeans.inertia_)

# Plot the elbow plot
plt.figure(figsize=(8, 6))
plt.plot(range(1, 11), inertia, marker='o', color='blue')
plt.title('Elbow Plot for Optimal k')
plt.xlabel('Number of Clusters (k)')
plt.ylabel('Inertia')
plt.xticks(range(1, 11))
plt.axvline(x=3, color='magenta', linestyle='--', label='Optimal k')
plt.legend()
plt.show()

# Apply PCA for dimensionality reduction
pca = PCA(n_components=6)  # You can choose the number of components
X_train_pca = pca.fit_transform(X_train_scaled)
X_test_pca = pca.transform(X_test_scaled)

# Create and train models with PCA-reduced data
models = {
    "SVM": SVC(kernel='linear', C=1.0),
    "Logistic Regression": LogisticRegression(),
    "Random Forest": RandomForestClassifier(n_estimators=100, random_state=42)
}

accuracies = {}

for model_name, model in models.items():
    model.fit(X_train_pca, y_train)
    y_pred = model.predict(X_test_pca)
    accuracy = accuracy_score(y_test, y_pred)
    accuracies[model_name] = accuracy

# Print accuracies for each model
for model_name, accuracy in accuracies.items():
    print(f"Accuracy - {model_name}:", accuracy)

# Plot the impact of PCA on explained variance ratio
explained_variance_ratio = pca.explained_variance_ratio_
plt.bar(range(1, len(explained_variance_ratio) + 1), explained_variance_ratio, color='purple')
plt.xlabel("Principal Components")
plt.ylabel("Explained Variance Ratio")
plt.title("Impact of PCA on Explained Variance Ratio")
plt.show()
