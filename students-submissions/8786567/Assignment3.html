def plotSilhouetteScores(data, title):
    silhouetteScores = {}
    kRange = range(2, 11)

    for k in kRange:
        kMeans = KMeans(n_clusters=k, random_state=42)
        kMeans.fit(data)
        silhouetteScores[k] = silhouette_score(data, kMeans.labels_)

    plt.plot(list(silhouetteScores.keys()), list(silhouetteScores.values()))
    plt.xlabel('Clusters')
    plt.ylabel('Silhouette Score')
    plt.title(title)
    plt.show()

    print("Silhouette Scores:")
    for k, score in silhouetteScores.items():
        print(f"Number of clusters: {k}, Silhouette Score: {score}")

def plotElbow(data, title):
    inertiaValues = []
    kRange = range(2, 11)

    for k in kRange:
        kMeans = KMeans(n_clusters=k, random_state=42)
        kMeans.fit(data)
        inertiaValues.append(kMeans.inertia_)

    plt.plot(kRange, inertiaValues)
    plt.xlabel('Clusters')
    plt.ylabel('Inertia')
    plt.title(title)
    plt.show()

    print("Inertia Values:")
    for k, inertia in zip(kRange, inertiaValues):
        print(f"Number of clusters: {k}, Inertia: {inertia}")

import numpy as np
import pandas as pd
from sklearn.preprocessing import StandardScaler, LabelEncoder
from sklearn.decomposition import PCA
from sklearn.model_selection import train_test_split
from sklearn.metrics import accuracy_score
from sklearn.svm import SVC
from sklearn.linear_model import LogisticRegression
from sklearn.ensemble import RandomForestClassifier

heartData = pd.read_csv('heart.csv')
print(heartData.head())

def filterOutliers(data, threshold=3, excludedCols=[]):
    outliersRemoved = {}

    for col in data.columns:
        if col in excludedCols:
            continue

        mean = np.mean(data[col])
        std = np.std(data[col])
        zScores = np.abs((data[col] - mean) / std)

        outliers = (zScores > threshold) | (zScores < -threshold)
        numRemoved = np.sum(outliers)
        outliersRemoved[col] = numRemoved

        data = data[~outliers]

    data.reset_index(drop=True, inplace=True)
    print("Number of removed outliers:", outliersRemoved)
    return data

def identifyBinaryFeatures(data, threshold=150):
    uniqueCounts = data.nunique()
    median = np.median(uniqueCounts)
    medianAbsoluteDeviation = np.median(np.abs(uniqueCounts - median))
    modifiedZScores = 0.6745 * (uniqueCounts - median) / medianAbsoluteDeviation
    return uniqueCounts[modifiedZScores < threshold].index.tolist()

binaryColumns = identifyBinaryFeatures(heartData, threshold=20)
binaryColumns.append('HeartDisease')  

filteredData = filterOutliers(heartData, excludedCols=binaryColumns)

def encodeLabels(data, columnsToEncode):
    encodedData = data.copy()
    labelEncoder = LabelEncoder()
    for col in columnsToEncode:
        encodedData[col] = labelEncoder.fit_transform(data[col])
    return encodedData

encodedData = encodeLabels(filteredData, binaryColumns)

def scaleFeatures(data, excludedCols=[]):
    scaledData = data.copy()
    columnsToScale = [col for col in scaledData.columns if col not in excludedCols]
    scaler = StandardScaler()
    scaledFeatures = scaler.fit_transform(scaledData[columnsToScale])
    scaledData[columnsToScale] = scaledFeatures
    return scaledData

scaledData = scaleFeatures(encodedData, binaryColumns)

def splitDataset(data, testSize=0.2):
    X = data.drop(data.columns[-1], axis=1)
    y = data[data.columns[-1]]
    XTrain, XTest, yTrain, yTest = train_test_split(X, y, test_size=testSize)
    return XTrain, XTest, yTrain, yTest

def evaluateAccuracy(yTest, yPred):
    accuracy = accuracy_score(yTest, yPred)
    print("Accuracy:", accuracy)

def evaluateModels(data, models):
    XTrain, XTest, yTrain, yTest = splitDataset(data)
    for model in models:
        print('\n', type(model).__name__, '\n')
        model.fit(XTrain, yTrain)
        yPred = model.predict(XTest)
        evaluateAccuracy(yTest, yPred)

def performPCA(data, explainedThreshold=0.95):
    X = data.drop(data.columns[-1], axis=1)
    y = data[data.columns[-1]]
    pca = PCA()
    transformedData = pca.fit_transform(X)
    explainedVariance = np.cumsum(pca.explained_variance_ratio_)
    nComponents = np.argmax(explainedVariance >= explainedThreshold) + 1
    pca = PCA(n_components=nComponents)
    transformedData = pca.fit_transform(X)
    transformedDF = pd.DataFrame(transformedData, columns=[f"component_{i+1}" for i in range(nComponents)])
    transformedDF['HeartDisease'] = y.values 
    return transformedDF

modelsToEvaluate = [
    SVC(),
    LogisticRegression(max_iter=1000),
    RandomForestClassifier()
]

evaluateModels(scaledData, modelsToEvaluate)

heartPCA = performPCA(scaledData)
evaluateModels(heartPCA, modelsToEvaluate)
