def k_fold_cross_validation(X, y, k):
    # Create KFold cross-validation object
    kf = KFold(n_splits=k, shuffle=True, random_state=56)
    # Perform cross-validation
    scores = cross_val_score(model, X, y, cv=kf)
    return scores
